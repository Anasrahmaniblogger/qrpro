        function customAlert(message) {
            return new Promise((resolve) => {
                const modalId = 'custom-alert-modal-' + Math.random().toString(36).substring(2, 9);
                const modalHtml = `
                    <div id="${modalId}" class="fixed inset-0 bg-slate-900/60 backdrop-blur-sm flex items-center justify-center z-[9999] p-4" style="animation: fadeIn 0.2s ease-out;">

                                <h3 class="text-lg font-bold text-slate-900">Notification</h3>
                                <p class="text-sm text-slate-600 mt-2 whitespace-pre-wrap leading-relaxed">${message}</p>
                            </div>
                            <button id="${modalId}-btn" class="w-full py-2.5 bg-blue-600 hover:bg-blue-700 text-white font-bold rounded-xl text-sm transition-all shadow-md active:scale-95">
                                Okay
                            </button>

                const btn = document.getElementById(modalId + '-btn');
                
                btn.onclick = () => {
                    modal.remove();
                    resolve();

        // Initialize App
        async function init() {
            // Listen for OAuth success messages from popups
            window.addEventListener('message', async (event) => {
                if (event.origin !== window.location.origin) return;
                
                if (event.data?.type === 'SUPABASE_AUTH_SUCCESS') {
                    console.log("OAuth success message received in main frame.");
                    if (supabaseClient) {
                        const { data: { session } } = await supabaseClient.auth.getSession();
                        if (session?.user) {
                            updateAuthUI(session.user);
                        }
                    }
                } else if (event.data?.type === 'SUPABASE_AUTH_ERROR') {
                    alert(event.data.error || "Authentication failed.");
                }
            });

            // Sync auth state across same-origin frames/popups via localStorage events
            window.addEventListener('storage', async (event) => {
                if (event.key && event.key.includes('auth-token')) {
                    if (supabaseClient) {
                        const { data: { session } } = await supabaseClient.auth.getSession();
                        updateAuthUI(session?.user);
                    }
                }
            });

            // Check if we are inside a popup and have an auth error in parameters
            if (window.opener && window.opener !== window) {
                const hasErrorParams = window.location.hash.includes('error=') || window.location.search.includes('error=');
                if (hasErrorParams) {
                    try {
                        window.opener.postMessage({ type: 'SUPABASE_AUTH_ERROR', error: 'Authentication failed or was cancelled.' }, window.location.origin);
                    } catch (e) {}
                    setTimeout(() => window.close(), 400);
                    return;
                }
            }

            if(SUPABASE_URL !== 'YOUR_SUPABASE_URL') {
                supabaseClient = window.supabase.createClient(SUPABASE_URL, SUPABASE_KEY);
                await checkUser();
                await seedSidebarMenu();
            } else {
                console.warn("Supabase not configured. Running in preview mode.");
            }
            await loadQRTemplates();
            await loadPricingPlans();
            await loadHeroImage();
            loadSidebar();
            
            // Generate initial QR
            generateQR();
        }

        // Helper to parse profile data and handle JSON-encoded fallback for missing DB columns
        function parseProfileData(profile) {
            let fullName = profile.full_name || '';
            let surname = profile.surname || '';
            let phone = profile.phone || '';
            let company = profile.company || '';
            let location = profile.location || '';
            let website = profile.website || '';
            let bio = profile.bio || '';
            
            // If full_name is a JSON string, extract the structured first_name, surname, phone and extra details
            if (fullName && fullName.trim().startsWith('{') && fullName.trim().endsWith('}')) {
                try {
                    const parsed = JSON.parse(fullName);
                    fullName = parsed.first_name || parsed.full_name || '';
                    surname = parsed.surname || '';
                    phone = parsed.phone || '';
                    company = parsed.company || '';
                    location = parsed.location || '';
                    website = parsed.website || '';
                    bio = parsed.bio || '';
                } catch (e) {
                    console.error("Failed to parse JSON full_name fallback:", e);
                }
            }
            
            return {
                fullName: fullName,
                surname: surname,
                phone: phone,
                company: company,
                location: location,
                website: website,
                bio: bio
            };
        }

        async function fetchUserProfile(userId) {
            if(!supabaseClient) {
                // Simulated / Mock profile loading for local testing
                let localProfile = JSON.parse(localStorage.getItem('profile_' + userId));
                if(!localProfile) {
                    localProfile = {
                        subscription_tier: 'FREE',
                        full_name: currentUser ? (currentUser.user_metadata?.full_name || '') : '',
                        surname: '',
                        email: currentUser ? (currentUser.email || '') : '',
                        phone: '',
                        company: '',
                        location: '',
                        website: '',
                        bio: '',
                        qr_generated_today: 0,
                        qr_limit: 2,
                        logos_generated_today: 0,
                        logos_limit: 1
                    };
                    localStorage.setItem('profile_' + userId, JSON.stringify(localProfile));
                }
                userProfile = localProfile;
                
                // Load local purchases
                let localPurchases = JSON.parse(localStorage.getItem('purchases_' + userId));
                if (Array.isArray(localPurchases)) {
                    userPurchasesCache = localPurchases;
                } else {
                    userPurchasesCache = [];
                }
                
                updateSubscriptionUI();
                populateProfileSettings();
                renderTemplatesGallery();
                return;
            }

            try {
                const { data, error } = await supabaseClient
                    .from('profiles')
                    .select('*')
                    .eq('id', userId)
                    .single();
                
                if (data) {
                    const profileDetails = parseProfileData(data);
                    userProfile = {
                        subscription_tier: data.subscription_tier || 'FREE',
                        full_name: profileDetails.fullName || '',
                        surname: profileDetails.surname || '',
                        email: data.email || (currentUser ? currentUser.email : ''),
                        phone: profileDetails.phone || '',
                        company: profileDetails.company || '',
                        location: profileDetails.location || '',
                        website: profileDetails.website || '',
                        bio: profileDetails.bio || '',
                        qr_generated_today: data.qr_generated_today || 0,
                        qr_limit: data.subscription_tier === 'PRO' ? Infinity : 2,
                        logos_generated_today: data.logos_generated_today || 0,
                        logos_limit: data.subscription_tier === 'PRO' ? Infinity : 1,
                        trial_start_date: data.trial_start_date
                    };

                    // Check trial expiry
                    if(userProfile.subscription_tier === 'FREE' && userProfile.trial_start_date) {
                        const trialStart = new Date(userProfile.trial_start_date);
                        const now = new Date();
                        const diffTime = Math.abs(now - trialStart);
                        const diffDays = Math.ceil(diffTime / (1000 * 60 * 60 * 24)); 
                        if(diffDays > 7) {
                            userProfile.trial_expired = true;
                            userProfile.qr_limit = 0;
                            userProfile.logos_limit = 0;
                        }
                    }
                } else {
                    // Profile does not exist, initialize blank state to trigger mandatory collection
                    userProfile = {
                        subscription_tier: 'FREE',
                        full_name: currentUser ? (currentUser.user_metadata?.full_name || '') : '',
                        surname: '',
                        email: currentUser ? (currentUser.email || '') : '',
                        phone: '',
                        company: '',
                        location: '',
                        website: '',
                        bio: '',
                        qr_generated_today: 0,
                        qr_limit: 2,
                        logos_generated_today: 0,
                        logos_limit: 1
                    };
                }
            } catch (err) {
                console.error("Error fetching profile from Supabase:", err);
            }

            // Check if mandatory details are missing
            checkMandatoryProfile();

            // Fetch purchases
            try {
                const { data: purchases } = await supabaseClient.from('template_purchases').select('template_id').eq('user_id', userId);
                if(purchases) {
                    userPurchasesCache = purchases.map(p => p.template_id);
                }
            } catch (err) {
                console.error("Error fetching purchases:", err);
            }

            updateSubscriptionUI();
            populateProfileSettings();
            renderTemplatesGallery();
        }

        async function saveUserProfile() {

            const surnameVal = document.getElementById('profile-surname-input').value.trim();
            const phoneVal = document.getElementById('profile-phone-input').value.trim();

            const bioVal = document.getElementById('profile-bio-input').value.trim();
            const emailVal = document.getElementById('profile-email-input').value.trim();

                return;

            const originalText = btn.innerHTML;
            btn.disabled = true;
            btn.innerHTML = '<i class="fas fa-spinner fa-spin mr-2"></i> Saving...';

                try {
                    // Check if profile exists
                    const { data: profileCheck } = await supabaseClient
                        .from('profiles')
                        .select('id')
                        .eq('id', currentUser.id)
                        .single();
                        
                    let error = null;
                    const updatePayload = {
                        full_name: nameVal,
                        surname: surnameVal,
                        phone: phoneVal,
                        company: companyVal,
                        location: locationVal,
                        website: websiteVal,
                        bio: bioVal
                    };
                    
                    if (profileCheck) {
                        const { error: err } = await supabaseClient
                            .from('profiles')
                            .update(updatePayload)
                            .eq('id', currentUser.id);
                        error = err;
                    } else {
                        const { error: err } = await supabaseClient
                            .from('profiles')
                            .insert([{
                                id: currentUser.id,
                                email: emailVal,
                                ...updatePayload
                            }]);
                        error = err;
                    }
                    
                    if (error) {
                        // Missing columns fallback - save inside full_name as structured JSON

                    // Update state

                    userProfile.surname = surnameVal;
                    userProfile.phone = phoneVal;
                    userProfile.company = companyVal;
                    userProfile.location = locationVal;
                    userProfile.website = websiteVal;

                } catch (err) {

        function canUserGenerateQR(templateId = null, hasBgImage = false) {
            if (userProfile.subscription_tier === 'PRO') {
                return { allowed: true };
            }

            if (userProfile.trial_expired) {
                return { allowed: false, reason: 'trial_expired' };
            }

            if (hasBgImage && userProfile.qr_generated_today >= 2) {
                return { allowed: false, reason: 'daily_limit' };
            }

            if (templateId) {
                const tmpl = qrTemplatesCache.find(t => t.id === templateId);
                if (tmpl) {
                    if (tmpl.is_premium) {
                        return { allowed: false, reason: 'requires_pro' };
                    }
                }
            }

            if (userProfile.qr_generated_today >= userProfile.qr_limit) {
                return { allowed: false, reason: 'daily_limit' };
            }
            
            return { allowed: true };
        }

        async function selectTemplate(template) {
            const access = canUserGenerateQR(template.id);
            
            if(!access.allowed) {
                if (access.reason === 'requires_pro') {
                    const proceed = await customConfirm(`This template requires a PRO subscription. Upgrade now to unlock all templates?`);
                    if(proceed) {
                        handleDirectUpgrade();
                    }
                    return;
                } else if (access.reason === 'requires_purchase') {
                    initiateCashfreePayment(template);
                    return;
                } else if (access.reason === 'trial_expired') {
                    const proceed = await customConfirm("Your 7-day free trial has expired. Upgrade to Premium now to use premium templates?");
                    if(proceed) {
                        handleDirectUpgrade();
                    }
                    return;
                }
                // If it's just daily_limit, we can still let them preview the template,
                // saveQR will block the actual generation.
            }

            // Apply design
            if(template.design) {
                if(template.design.dotsType) document.getElementById('qr-dots-type').value = template.design.dotsType;
                if(template.design.cornersSquareType) document.getElementById('qr-corners-square-type').value = template.design.cornersSquareType;
                if(template.design.cornersDotType) document.getElementById('qr-corners-dot-type').value = template.design.cornersDotType;
                if(template.design.fg) {
                    document.getElementById('qr-color-fg').value = template.design.fg;
                    document.getElementById('qr-color-fg').nextElementSibling.value = template.design.fg;
                }
                if(template.design.bg) {
                    document.getElementById('qr-color-bg').value = template.design.bg;
                    document.getElementById('qr-color-bg').nextElementSibling.value = template.design.bg;
                }
                generateQR();
            }
        }

        async function initiateCashfreePayment(template) {
            if(!currentUser) {
                alert("Please log in to purchase.");
                return;
            }
            if (!userProfile.full_name || !userProfile.surname || !userProfile.phone) {
                alert("Please complete your profile details (First Name, Surname, Phone) in the Profile Section before proceeding to payment.");
                navigate('dashboard');
                showDashSection('profile');
                return;
            }

            const isMockMode = !supabaseClient || !SUPABASE_URL || SUPABASE_URL === 'YOUR_SUPABASE_URL';
            if (isMockMode) {
                await runMockPayment(template);
                return;
            }

            try {
                // 1. Call Backend to create Cashfree Order (Vercel API)
                const response = await fetch(`/api/create-order`, {
                    method: 'POST',
                    headers: {
                        'Content-Type': 'application/json',
                        'Authorization': `Bearer ${supabaseClient ? (await supabaseClient.auth.getSession()).data.session?.access_token : ''}`
                    },
                    body: JSON.stringify({
                        type: 'template',
                        template_id: template.id,
                        amount: template.price,
                        customer_id: currentUser.id,
                        customer_email: currentUser.email,
                        customer_name: userProfile.full_name || 'User',
                        customer_phone: userProfile.phone || '9999999999'
                    })
                });

                if(!response.ok) throw new Error("Backend connection failed");

                const { payment_session_id, order_id } = await response.json();

                // 2. Open Cashfree Checkout
                if(cashfree) {
                    let checkoutOptions = {
                        paymentSessionId: payment_session_id,
                        redirectTarget: "_modal",
                    };
                    cashfree.checkout(checkoutOptions).then(function (result) {
                        if(result.error){
                            alert("Payment failed: " + result.error.message);
                        }
                        if(result.paymentDetails){
                            verifyPayment(order_id, template.id, 'template');
                        }
                    });
                } else {
                    throw new Error("Cashfree SDK not initialized");
                }
            } catch (err) {
                console.warn("Backend edge function not deployed. Using mock preview flow.", err);
                await runMockPayment(template);
            }
        }

                headers: {
                    'Content-Type': 'application/json',
                    'Authorization': `Bearer ${supabaseClient ? (await supabaseClient.auth.getSession()).data.session?.access_token : ''}`
                },
                body: JSON.stringify({ order_id: orderId, type: type, template_id: templateId })
            });

            const data = await response.json();

                alert("Payment Verified! Features unlocked.");

        async function signInWithGoogle() {
            if(!supabaseClient) {
                alert("Supabase not configured. To test the UI, we will simulate login.");
                updateAuthUI({ id: 'preview-user', email: 'test@example.com' });
                return;
            }
            
            try {
                // Redirect back to user.html

                const { data, error } = await supabaseClient.auth.signInWithOAuth({
                    provider: 'google',
                    options: {
                        redirectTo: redirectUri,
                        skipBrowserRedirect: true
                    }
                });
                
                if (error) {
                    throw error;
                }
                
                if (data?.url) {
                    // Open the authorization URL in a popup window
                    const width = 600;
                    const height = 700;
                    const left = (window.screen.width - width) / 2;
                    const top = (window.screen.height - height) / 2;
                    
                    const authWindow = window.open(
                        data.url,
                        'google_oauth_popup',
                        `width=${width},height=${height},top=${top},left=${left},resizable=yes,scrollbars=yes,status=yes`
                    );
                    
                    if (!authWindow) {
                        alert("Popup blocked! Please allow popups for this website, or open the app in a new tab to sign in with Google.");
                    }
                } else {
                    alert("Could not generate Google sign-in URL.");
                }
            } catch (err) {
                console.error("Google Sign In Error:", err);
                alert("Sign-in failed: " + (err.message || err));
            }
        }

        async function generateAIContent(type) {
            const prompt = document.getElementById(type === 'ai' ? 'ai-qr-prompt' : 'ai-logo-prompt').value;
            if (!prompt) return alert('Please enter a prompt!');
            
            const resultDiv = document.getElementById(type === 'ai' ? 'ai-qr-result' : 'ai-logo-result');
            resultDiv.innerText = 'Generating...';
            
            try {
                const { data: { session } } = await supabaseClient.auth.getSession();
                if (!session) throw new Error('Please log in first!');

                const response = await fetch(`${SUPABASE_URL}/functions/v1/generate-content`, {
                    method: 'POST',
                    headers: { 'Authorization': `Bearer ${session.access_token}`, 'Content-Type': 'application/json' },
                    body: JSON.stringify({ type, prompt })
                });
                const result = await response.json();
                
                if (result.error) throw new Error(result.error);
                
                resultDiv.innerHTML = `<img src="${result.result}" class="rounded-lg shadow-md w-full">`;

            }

        function generateQR() {
            const container = document.getElementById("qrcode-canvas");
            
            const text = document.getElementById("qr-data").value || "https://qrpro.com";
            const colorDark = document.getElementById("qr-color-fg").value;
            const colorLight = document.getElementById("qr-color-bg").value;
            const correctLevel = document.getElementById("qr-error").value;
            
            const dotsType = document.getElementById("qr-dots-type")?.value || "square";
            const cornersSquareType = document.getElementById("qr-corners-square-type")?.value || "square";
            const cornersDotType = document.getElementById("qr-corners-dot-type")?.value || "square";
            const margin = parseInt(document.getElementById("qr-margin")?.value || "10", 10);
            const maintainAspectRatio = document.getElementById("qr-logo-aspect-ratio")?.checked || false;

            const options = {
                width: 300,
                height: 300,
                data: text,
                margin: margin,
                qrOptions: {
                    typeNumber: 0,
                    mode: "Byte",
                    errorCorrectionLevel: correctLevel
                },
                imageOptions: {
                    hideBackgroundDots: maintainAspectRatio,
                    imageSize: 0.4,
                    margin: 5
                },
                dotsOptions: {
                    type: dotsType,
                    color: colorDark
                },
                backgroundOptions: {
                    color: colorLight
                },
                cornersSquareOptions: {
                    type: cornersSquareType,
                    color: colorDark
                },
                cornersDotOptions: {
                    type: cornersDotType,
                    color: colorDark
                }
            };
            
            if (currentLogoUrl) {
                options.image = currentLogoUrl;
            }

            if (!qrCodeInstance) {
                qrCodeInstance = new QRCodeStyling(options);
                container.innerHTML = ""; // Clear existing
                qrCodeInstance.append(container);
            } else {
                qrCodeInstance.update(options);
            }
            
            // Handle background image
            const bgImageInput = document.getElementById("qr-bg-image");
            if (bgImageInput.files && bgImageInput.files[0]) {
                const reader = new FileReader();
                reader.onload = (e) => {
                    container.style.backgroundImage = `url(${e.target.result})`;
                    container.style.backgroundSize = "cover";
                    container.style.backgroundPosition = "center";
                };
                reader.readAsDataURL(bgImageInput.files[0]);
            } else {
                container.style.backgroundImage = "none";
            }
        }

        async function downloadQR() {
            if(qrCodeInstance) {
                if(userProfile.trial_expired) {
                    const proceed = await customConfirm("Your 7-day free trial has expired. Upgrade to Premium now to download QR codes?");
                    if(proceed) {
                        handleDirectUpgrade();
                    } else {
                        navigate('dashboard');
                        showDashSection('subscription');
                    }
                    return;
                }

                let currentWidth = 300;
                let currentHeight = 300;
                
                if (userProfile.subscription_tier !== 'PRO') {
                    const proceed = await customConfirm("Free Plan export is standard resolution (300px). Would you like to Upgrade to Premium for High-Resolution (1000px) exports and instant download? Click 'Cancel' to continue standard resolution download.");
                    if (proceed) {
                        handleDirectUpgrade();
                        return;
                    }
                } else {
                    currentWidth = 1000;
                    currentHeight = 1000;
                }

                // Temporary update for download
                qrCodeInstance.update({ width: currentWidth, height: currentHeight });
                
                qrCodeInstance.download({ name: "QR_Pro_Code", extension: "png" }).then(() => {
                    // Revert back to preview size
                    qrCodeInstance.update({ width: 300, height: 300 });
                });
            }
        }

        async function saveQR() {
            if(!currentUser) {
                alert("Please log in to save your QR codes.");
                navigate('auth');
                return;
            }
            
            const hasBgImage = !!document.getElementById("qr-bg-image")?.files[0];
            const access = canUserGenerateQR(null, hasBgImage);
            if (!access.allowed) {
                let msg = "";
                if (access.reason === 'trial_expired') {
                    msg = "Your 7-day free trial has expired. Upgrade to Premium now to continue generating QR codes?";
                } else if (access.reason === 'daily_limit') {
                    msg = "You have reached your daily QR generation limit. Upgrade to Premium now for unlimited generations?";
                } else if (access.reason === 'requires_pro') {
                    msg = "This feature requires a PRO subscription. Upgrade now to unlock?";
                }
                const proceed = await customConfirm(msg);
                if (proceed) {
                    handleDirectUpgrade();
                } else {
                    navigate('dashboard');
                    showDashSection('subscription');
                }
                return;
            }
            
            if(currentLogoUrl && userProfile.logos_generated_today >= userProfile.logos_limit) {
                const proceed = await customConfirm("You have reached your daily logo upload limit. Upgrade to Premium for unlimited logo uploads?");
                if(proceed) {
                    handleDirectUpgrade();
                } else {
                    navigate('dashboard');
                    showDashSection('subscription');
                }
                return;
            }
            
            if(!supabaseClient) {
                alert("Preview mode: QR generated! (Supabase not configured)");
                userProfile.qr_generated_today += 1;
                if(currentLogoUrl) userProfile.logos_generated_today += 1;
                updateSubscriptionUI();
                return;
            }
            
            const btn = event.currentTarget;
            const originalText = btn.innerHTML;
            btn.innerHTML = '<i class="fas fa-spinner fa-spin"></i> Saving...';
            
            // Build payload
            const payload = {
                user_id: currentUser.id,
                name: "My QR Code",
                type: "URL",
                data: { url: document.getElementById("qr-data").value },
                design: {
                    fg: document.getElementById("qr-color-fg").value,
                    bg: document.getElementById("qr-color-bg").value,
                    error: document.getElementById("qr-error").value,
                    logo: currentLogoUrl ? "included" : null
                }
            };
            
            const { data, error } = await supabaseClient.from('qr_codes').insert([payload]);
            
            btn.innerHTML = originalText;
            
            if(error) {
                alert(error.message);
            } else {
                alert("QR Code Saved Successfully!");
                // Increment local counter and update UI
                userProfile.qr_generated_today += 1;
                if(currentLogoUrl) userProfile.logos_generated_today += 1;
                
                // Update profile in DB
                await supabaseClient
                    .from('profiles')
                    .update({ 
                        qr_generated_today: userProfile.qr_generated_today,
                        logos_generated_today: userProfile.logos_generated_today
                    })
                    .eq('id', currentUser.id);
                    
                updateSubscriptionUI();
            }
        }

        async function generateLogo() {
            if(!currentUser) {
                alert("Please log in to generate logos.");
                navigate('auth');
                return;
            }

            const prompt = document.getElementById('logo-prompt').value;
            const brand = document.getElementById('logo-brand-name').value;
            
            if (!brand || !prompt) {
                alert("Please provide both a brand name and a description for your logo.");
                return;
            }

            if (userProfile.subscription_tier !== 'PRO' && userProfile.logos_generated_today >= userProfile.logos_limit) {
                const proceed = await customConfirm("You have reached your daily logo generation limit. Upgrade to Premium for unlimited logo generations?");
                if(proceed) {
                    handleDirectUpgrade();
                } else {
                    navigate('dashboard');
                    showDashSection('subscription');
                }
                return;
            }

            const emptyState = document.getElementById('logo-empty-state');
            const loadingState = document.getElementById('logo-loading-state');
            const resultState = document.getElementById('logo-result-state');
            const previewImage = document.getElementById('logo-preview-image');
            const downloadBtn = document.getElementById('btn-download-logo');
            const generateBtn = document.getElementById('btn-generate-logo');

            // Show loading
            emptyState.classList.add('hidden');
            resultState.classList.add('hidden');
            loadingState.classList.remove('hidden');
            generateBtn.disabled = true;
            generateBtn.innerHTML = '<i class="fas fa-circle-notch fa-spin"></i> Generating...';

            // Simulate API call to AI generation service
            setTimeout(async () => {
                // Placeholder image for demo purposes
                const style = document.getElementById('logo-style').value;
                const colors = document.getElementById('logo-colors').value;
                
                // Using a placeholder image service to simulate generated logo
                // In a real app, this would be an API call to OpenAI/Midjourney/Gemini
                const seed = encodeURIComponent(`${brand} ${prompt} ${style} ${colors} logo`);
                const generatedUrl = `https://api.dicebear.com/9.x/shapes/svg?seed=${seed}&backgroundColor=transparent`;
                
                previewImage.src = generatedUrl;
                
                // Update state
                loadingState.classList.add('hidden');
                resultState.classList.remove('hidden');
                downloadBtn.disabled = false;
                generateBtn.disabled = false;
                generateBtn.innerHTML = '<i class="fas fa-wand-magic-sparkles"></i> Generate Another';
                
                // Increment counter
                userProfile.logos_generated_today += 1;
                updateSubscriptionUI();
                
                if (supabaseClient) {
                    await supabaseClient
                        .from('profiles')
                        .update({ logos_generated_today: userProfile.logos_generated_today })
                        .eq('id', currentUser.id);
                }
            }, 3500); // 3.5s delay to simulate generation
        }

        function downloadLogo() {
            const previewImage = document.getElementById('logo-preview-image');
            if (!previewImage || !previewImage.src) return;
            
            fetch(previewImage.src)
                .then(response => response.blob())
                .then(blob => {
                    const url = window.URL.createObjectURL(blob);
                    const a = document.createElement('a');
                    a.style.display = 'none';
                    a.href = url;
                    const brand = document.getElementById('logo-brand-name').value || 'logo';
                    a.download = `${brand.replace(/\s+/g, '_').toLowerCase()}_generated.svg`;
                    document.body.appendChild(a);
                    a.click();
                    window.URL.revokeObjectURL(url);
                    document.body.removeChild(a);
                })
                .catch(err => {
                    alert("Error downloading logo. Please try again.");
                });
        }

        function toggleFaq(btn) {
            const content = btn.nextElementSibling;
            const icon = btn.querySelector('i');
            
            // Optional: Close other FAQs
            document.querySelectorAll('#faq-accordion > div > div').forEach(div => {
                if(div !== content && div.style.maxHeight) {
                    div.style.maxHeight = null;
                    div.previousElementSibling.querySelector('i').classList.remove('rotate-180');
                }
            });

            if (content.style.maxHeight) {
                content.style.maxHeight = null;
                icon.classList.remove('rotate-180');
            } else {
                content.style.maxHeight = content.scrollHeight + "px";
                icon.classList.add('rotate-180');
            }
        }
